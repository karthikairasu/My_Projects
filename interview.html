  1. Call
  2. With the call() method, you can write a method that can be used on different objects.

	          const person = {
		  fullName: function(city, country) {

		  }
		}

		const person1 = {
		  firstName:"John",

	      function Company(name, age){
		  this.name = name;
		  this.age = age;
		}

		function Student1(name, age){
		  Company.call(this, name, age);
		  this.sec = "A";
		}

		function Student2(name, age){
		  Company.call(this, name, age); 
		  this.sec = "B";
		}

		var res1 = new Student1("siva", 25);
		console.log(res1.name, res1.age, res1.sec);  

		var res1 = new Student2("logi", 28);
		console.log(res1.name, res1.age, res1.sec);
